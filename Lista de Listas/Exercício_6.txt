    }

    Elemento* n1 = *L1;
    /** Percorrendo a lista 1 e interligando o final dela com o
        inicio da lista 2 **/
    while(n1 != NULL){
        if(n1->prox == NULL){
            n1->prox = (*L2);
            break;
        }
        n1 = n1->prox;
    }
    /// Criando uma lista que recebera a uniao dessas listas
    Lista* L3 = cria_lista();

    libera_lista(L1);

}
